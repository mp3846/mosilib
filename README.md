## Install

```bash
npm install --save mosilib
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'mosilib'
import 'mosilib/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
